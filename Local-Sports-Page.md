1. PHP Proxy Script Example (feeds-proxy.php)

<?php
/**
 * feeds-proxy.php
 * Production-ready server-side proxy for Redlands & Bay Islands sports feeds
 * Converts RSS/XML to JSON for dynamic front-end rendering
 * Caching enabled for performance (12 hours)
 * Author: AiAscended / TheBayIslands.au 2026
 */

// --- CONFIGURATION ---
$feeds = [
    'rugby' => 'https://www.redlandsrugbyleague.com.au/events-feed.xml',
    'netball' => 'https://www.redlandsnetball.com.au/wspCompetitions.aspx?rss=true',
    'soccer' => 'https://redlandsaustralia.teamapp.com/api/events', // TeamApp JSON API
    'council' => 'https://www.redland.qld.gov.au/News-events-and-have-your-say/Events-and-whats-on/Whats-On-calendar?rss=true'
];
$cacheDir = __DIR__ . '/cache';
$cacheTime = 43200; // 12 hours in seconds

// Ensure cache directory exists
if(!is_dir($cacheDir)) mkdir($cacheDir, 0755, true);

// Determine feed to serve
$type = isset($_GET['type']) ? $_GET['type'] : 'rugby';
if(!isset($feeds[$type])) { http_response_code(400); echo json_encode(['error'=>'Invalid feed type']); exit; }

$cacheFile = "$cacheDir/$type.json";

// Serve cache if valid
if(file_exists($cacheFile) && (time() - filemtime($cacheFile)) < $cacheTime) {
    header('Content-Type: application/json');
    echo file_get_contents($cacheFile);
    exit;
}

// Fetch feed
$feedUrl = $feeds[$type];
$raw = @file_get_contents($feedUrl);
if(!$raw) { http_response_code(500); echo json_encode(['error'=>"Feed fetch failed: $feedUrl"]); exit; }

// Parse feed
$events = [];
if(strpos($feedUrl, '.xml') !== false || strpos($feedUrl, 'rss=true') !== false) {
    // RSS/XML
    $xml = simplexml_load_string($raw, 'SimpleXMLElement', LIBXML_NOCDATA);
    foreach($xml->channel->item as $item) {
        $startDate = (string)$item->pubDate ?? (string)$item->date ?? date('c');
        $events[] = [
            'title' => (string)$item->title,
            'startDate' => date('c', strtotime($startDate)),
            'url' => (string)$item->link,
            'location' => (string)$item->location ?? '',
            'locationName' => (string)$item->location ?? '',
            'locationAddress' => (string)$item->location ?? '',
            'sportIcon' => match($type) {
                'rugby' => 'üèâ',
                'netball' => 'ü§æ‚Äç‚ôÄÔ∏è',
                'soccer' => '‚öΩ',
                'council' => 'üéâ',
                default => 'üìå'
            }
        ];
    }
} else {
    // JSON API (TeamApp)
    $jsonData = json_decode($raw,true);
    if($jsonData) {
        foreach($jsonData['events'] ?? [] as $ev) {
            $events[] = [
                'title' => $ev['title'] ?? 'Event',
                'startDate' => $ev['startDate'] ?? $ev['date'] ?? date('c'),
                'url' => $ev['url'] ?? '#',
                'location' => $ev['location'] ?? '',
                'locationName' => $ev['location'] ?? '',
                'locationAddress' => $ev['location'] ?? '',
                'sportIcon' => match($type) {
                    'soccer' => '‚öΩ',
                    default => 'üìå'
                }
            ];
        }
    }
}

// Save to cache
file_put_contents($cacheFile, json_encode($events, JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE));

// Output JSON
header('Content-Type: application/json');
echo json_encode($events, JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE);
exit;
