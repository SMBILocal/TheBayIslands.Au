### Step 3: Create Sports Club Component

**File:** `components/SportsClubCard.tsx`

```typescript
'use client';

interface SportsClubCardProps {
  name: string;
  sport: string;
  location: string;
  icon: string;
  divisions?: string[];
}

export default function SportsClubCard({
  name,
  sport,
  location,
  icon,
  divisions = [],
}: SportsClubCardProps) {
  return (
    <div className="border rounded-lg p-6 hover:shadow-lg transition-shadow">
      <div className="flex items-start gap-4">
        <span className="text-4xl">{icon}</span>
        <div className="flex-1">
          <h3 className="text-xl font-bold mb-2">{name}</h3>
          <p className="text-gray-600 mb-2">{sport}</p>
          <p className="text-sm text-gray-500 mb-2">{location}</p>
          {divisions.length > 0 && (
            <div className="flex gap-2">
              {divisions.map((div) => (
                <span
                  key={div}
                  className="text-xs bg-blue-100 text-blue-800 px-2 py-1 rounded"
                >
                  {div}
                </span>
              ))}
            </div>
          )}
        </div>
      </div>
    </div>
  );
}
```

## Database Migrations (Run First)

Before implementing any features, run these database migrations:

```sql
-- Run this in Supabase SQL Editor

-- Articles table
CREATE TABLE IF NOT EXISTS articles (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  title TEXT NOT NULL,
  slug TEXT UNIQUE NOT NULL,
  content TEXT,
  excerpt TEXT,
  featured_image TEXT,
  author_id UUID REFERENCES users(id),
  category TEXT,
  tags TEXT[],
  status TEXT DEFAULT 'draft',
  published_at TIMESTAMP WITH TIME ZONE,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Community notices table
CREATE TABLE IF NOT EXISTS notices (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  title TEXT NOT NULL,
  content TEXT,
  demographic TEXT,
  category TEXT,
  user_id UUID REFERENCES users(id),
  status TEXT DEFAULT 'pending',
  expires_at TIMESTAMP WITH TIME ZONE,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Sports clubs table
CREATE TABLE IF NOT EXISTS sports_clubs (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name TEXT NOT NULL,
  sport TEXT,
  location TEXT,
  feed_url TEXT,
  icon TEXT,
  divisions TEXT[],
  contact_email TEXT,
  contact_phone TEXT,
  status TEXT DEFAULT 'active',
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- External feeds table
CREATE TABLE IF NOT EXISTS external_feeds (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name TEXT NOT NULL,
  url TEXT NOT NULL,
  type TEXT,
  category TEXT,
  last_fetched TIMESTAMP WITH TIME ZONE,
  status TEXT DEFAULT 'active',
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
```
