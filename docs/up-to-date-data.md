16.1 app/sitemap.ts

Dynamic, tier-aware, locality-aware sitemap

import { MetadataRoute } from "next"
import { prisma } from "@/lib/db"

const BASE_URL = "https://yourdomain.com" // CHANGE THIS

export default async function sitemap(): Promise<MetadataRoute.Sitemap> {
  const [businesses, suburbs, categories] = await Promise.all([
    prisma.business.findMany({
      select: {
        slug: true,
        updatedAt: true,
        tier: true
      }
    }),
    prisma.suburb.findMany({
      select: {
        slug: true,
        createdAt: true
      }
    }),
    prisma.category.findMany({
      select: {
        slug: true,
        createdAt: true
      }
    })
  ])

  const tierPriority: Record<string, number> = {
    FREE: 0.4,
    STARTER: 0.6,
    BOOST: 0.7,
    AUTHORITY: 0.8,
    LEAD_ENGINE: 0.9,
    PREMIUM: 1.0
  }

  const staticPages: MetadataRoute.Sitemap = [
    {
      url: `${BASE_URL}`,
      lastModified: new Date(),
      priority: 1.0
    },
    {
      url: `${BASE_URL}/pricing`,
      lastModified: new Date(),
      priority: 0.8
    },
    {
      url: `${BASE_URL}/add-business`,
      lastModified: new Date(),
      priority: 0.7
    }
  ]

  const suburbPages = suburbs.map(suburb => ({
    url: `${BASE_URL}/suburbs/${suburb.slug}`,
    lastModified: suburb.createdAt,
    priority: 0.8
  }))

  const categoryPages = categories.map(category => ({
    url: `${BASE_URL}/categories/${category.slug}`,
    lastModified: category.createdAt,
    priority: 0.7
  }))

  const suburbCategoryPages = suburbs.flatMap(suburb =>
    categories.map(category => ({
      url: `${BASE_URL}/${suburb.slug}/${category.slug}`,
      lastModified: new Date(),
      priority: 0.9
    }))
  )

  const businessPages = businesses.map(business => ({
    url: `${BASE_URL}/business/${business.slug}`,
    lastModified: business.updatedAt,
    priority: tierPriority[business.tier]
  }))

  return [
    ...staticPages,
    ...suburbPages,
    ...categoryPages,
    ...suburbCategoryPages,
    ...businessPages
  ]
}
